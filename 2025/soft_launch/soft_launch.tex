\hlcom{# Count the occurrences of "don't know" responses per respondent}

  \hlkwd{select}\hldef{(}\hlopt{-}\hldef{ResponseId)} \hlopt{%>%}  \hlcom{# Exclude the ResponseId column}
  \hlkwd{mutate_all}\hldef{(}\hlopt{~} \hldef{.} \hlopt{%in%} \hldef{dont_know_variations)} \hlopt{%>%}  \hlcom{# Check for "don't know" variations}

\hlcom{# Combine with ResponseId for reference}
\hldef{respondent_dont_know} \hlkwb{<-} \hlkwd{data.frame}\hldef{(}
  \hlkwc{ResponseId} \hldef{= dat.t}\hlopt{$}\hldef{ResponseId,}
  \hlkwc{DontKnowCount} \hldef{= dont_know_count}
\hldef{)}

  \hlkwd{arrange}\hldef{(}\hlkwd{desc}\hldef{(DontKnowCount))} \hlopt{%>%}
  \hlkwd{filter}\hldef{(DontKnowCount} \hlopt{>} \hlnum{0}\hldef{)}  \hlcom{# Optional: Exclude respondents with no "don't know" answers}
